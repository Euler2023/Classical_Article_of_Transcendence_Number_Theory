\documentclass{qbook}

% Enable makeidx for generating dynamic index corresponding to new page numbers
\usepackage{makeidx}
\usepackage{mathrsfs}
\usepackage{multicol}
\makeindex

% ==================================================================
% Custom Academic Footnote Symbols (Baker Style: Page-by-page starting with †)
% ==================================================================
\usepackage[perpage]{footmisc}
\makeatletter
\def\@fnsymbol#1{\ensuremath{\ifcase#1\or \dagger\or \ddagger\or \S\or \P\or \|\or \dagger\dagger\or \ddagger\ddagger\or \S\S\or \P\P\else *\fi}}
\makeatother
\renewcommand{\thefootnote}{\fnsymbol{footnote}}

% ==================================================================
% Academic Section, Theorem, and Equation Numbering (Baker Style)
% ==================================================================
% 1. Section numbering without Chapter prefix (e.g., "1" instead of "1.1")
\renewcommand{\thesection}{\arabic{section}}

% 2. Theorem and Lemma numbering
\newtheorem{theorem}{Theorem}[chapter]
\newtheorem{lemma}{Lemma}[chapter]
\renewcommand{\thelemma}{\arabic{lemma}}

% 3. Equation numbering resetting per chapter and without Chapter prefix (e.g., "(1)" instead of "(1.1)")
\makeatletter
\@addtoreset{equation}{chapter}
\makeatother
\renewcommand{\theequation}{\arabic{equation}}

% ==================================================================
% Custom hanglist environment for appendices (Original Papers / Further Publications)
% ==================================================================
\newenvironment{hanglist}{%
  \begin{list}{}{%
    \setlength{\leftmargin}{1.5em}%
    \setlength{\itemindent}{-1.5em}%
    \setlength{\itemsep}{3pt}%
    \setlength{\parsep}{0pt}%
  }%
}{\end{list}}

% 4. Autoref names (English)
\AtBeginDocument{%
  \def\chapterautorefname{Chapter}%
  \def\sectionautorefname{\S}%
  \def\subsectionautorefname{\S}%
  \def\equationautorefname{Equation}%
  \def\theoremautorefname{Theorem}%
  \def\lemmaautorefname{Lemma}%
  \def\definitionautorefname{Definition}%
}

% 5. Redefine theindex to prepend the author's intro note spanning full-width above the two columns
\makeatletter
\renewenvironment{theindex}{%
  \chapter*{\indexname}%
  \@mkboth{\MakeUppercase\indexname}{\MakeUppercase\indexname}%
  \thispagestyle{plain}%
  \vspace{1.5em}%
  \noindent\textmd{Entries for authors refer to places where either their name or one of their publications has been cited.}\vspace{1.5em}%
  \parindent\z@%
  \parskip\z@\@plus.3\p@\relax%
  \columnsep 35\p@%
  \columnseprule\z@%
  \let\item\@idxitem%
  \begin{multicols}{2}%
}{%
  \end{multicols}%
}
\makeatother
