% 4. Autoref names (Chinese)
\makeatletter
\AtBeginDocument{%
  \def\chapterautorefname{}%
  \def\sectionautorefname{}%
  \def\subsectionautorefname{}%
  \def\equationautorefname{公式}%
  \def\theoremautorefname{定理}%
  \def\lemmaautorefname{引理}%
  \def\definitionautorefname{定义}%
  %
  % Custom Chinese autoref format with prefix & suffix
  \let\oldautoref\autoref
  \gdef\autoref#1{%
    \edef\HyRef@label{#1}%
    \expandafter\HyRef@checkprefix\HyRef@label::\@nil{#1}%
  }%
  \gdef\HyRef@checkprefix#1:#2:#3\@nil#4{%
    \def\HyRef@temp{#1}%
    \def\HyRef@ch{ch}%
    \def\HyRef@sec{sec}%
    \def\HyRef@subsec{subsec}%
    \ifx\HyRef@temp\HyRef@ch%
      第\oldautoref{#4}~章%
    \else\ifx\HyRef@temp\HyRef@sec%
      第\oldautoref{#4}~节%
    \else\ifx\HyRef@temp\HyRef@subsec%
      第\oldautoref{#4}~小节%
    \else%
      \oldautoref{#4}%
    \fi\fi\fi%
  }%
}
\makeatother
